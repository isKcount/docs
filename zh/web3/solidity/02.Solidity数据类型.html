                            # Solidity数据类型

                            # 1.数据类型分类

                            # 1.1 值类型/基本类型

                            常用的基本数据类型分类有如下几种：

                            • 布尔(bool)
                            • 整形(int/uint)
                            • 地址(address)
                            • 定长字节数组
                            • 定长浮点(fixed/ufixed)
                            • 枚举类型(enum)
                            • 函数类型(function)

                            # 1.2 引用类型

                            常用的引用数据类型分类如下：

                            • 数组(Array) 不定长字节数组(bytes) 字符串(string)
                            • 结构体(Status)
                            • 映射(Mapping)

                            # 2. 数据类型详解

                            # 2.1 int和uint

                            int：表示有符号的整数

                            分类如下：

                            • int8 uint8
                            • int16 uint16
                            • int24 uint24
                            • int32 uint32
                            • ...
                            • int256 uint256

                            以8为递增序列，最多到256

                            int等价于int256 uint等价于uint256

                            不同表现形式的整形能够表示的数据的范围的公式为：

                            • 设位数为n
                            • int类型能表示的数据范围公式为： -2^(n-1) ~ 2^(n-1) -1
                            • uint类型能表示的数据范围公式为： 0 - 2^n-1
                            pragma solidity >=0.7.0 <0.9.0;

contract TestInt {
    int num1 = 256;
    int8 num2 = -100;
    int256 num3 = 1000;
    uint8 num4 = 10;

    // int类型
    function setNum1(int  _num1) public {
        num1 = _num1;
    }
    function getNum1() public view returns (int){
        return num1;
    }
    // int8类型
    function setNum2(int8  _num2) public {
        num2 = _num2;
    }
    function getNum2() public view returns (int8){
        return num2;
    }
    // int256类型
    function setNum3(int256  _num3) public {
        num3 = _num3;
    }
    function getNum3() public view returns (int256){
        return num3;
    }
    // uint8类型
    function setNum4(uint8  _num4) public {
        num4 = _num4;
    }
    function getNum4() public view returns (uint8){
        return num4;
    }
}

                            # 2.2 布尔型

                            布尔型表示真假： true或者false

                            pragma solidity >=0.7.0 <0.9.0;

contract TestBool {
    int a = 10;
    int b = 20;

    function checkInt() public returns(bool) {
    // 判断两个变量是否相等
        if (a == b) {
            return true;
        }else {
            return false;
        }
    } 
}

                            # 2.3 字节类型

                            字节类型bytes是计算机中最小的存储单位

                            • 一个byte = 8位
                            • 在solidity中，byte可以赋值为： 16进制的数，一个单引号引起来的单个或多个字符。
                            • 在solidity中，字节类型的byte,也被称之为定长字节数组
                            • 定长字节数组的长度不可改变

                            byte的表现形式

                            • bytes1 > bytes2 > bytes3 > bytes4 > ...... > bytest32
                            • bytes32 等价于int256位或者uint256位 32*8=256
                            • byte等价于bytes1

                            属性

                            • length: 表示这个字节数组的长度（只读）。若类型为byte1则返回1
                            • 允许使用下标来访问数据中保存的每8位的数据，下标0开始
                            pragma solidity >=0.7.0 <0.9.0;

contract TestByte {
    bytes1 b2 = 0x97;
    bytes3 b3 = 'abc';
    bytes32 b4 = 'abcdefg';
    
    // bytes1类型
    function setb2(bytes1 _b2) public {
        b2 = _b2;
    }
    function getb2() public view returns(bytes1){
        return b2;
    }
    // bytes2类型
    function setb3(bytes3 _b3) public {
        b3 = _b3;
    }
    function getb3() public view returns(bytes3){
        return b3;
    }
    // bytes32类型
    function setb4(bytes32 _b4) public {
        b4 = _b4;
    }
    function getb4() public view returns(bytes32){
        return b4;
    }    
}

                            # 2.4 字符串

                            Solidity同时使用双引号和单引号支持string字面量。它提供字符串作为数据类型类声明String类型的变量。

                            字符串的转移：

                            • \n 下一行
                            • \\ 反斜杠
                            • \' 单引号
                            • \" 双引号
                            • \b 退格
                            • \f 分页符
                            • \r 换行
                            • \t tab空格
                            • \v 垂直竖表符

                            字节到字符串的转换

                            bytes memory bstr = new bytes(10);
string message = string(bstr);

                            实例：

                            pragma solidity >=0.7.0 <0.9.0;

contract TestByte2 {
    
    string name = "tongxuejava";
    
    function getLength() public view returns(uint){
        // return name.length; 不能够直接的获取string的长度
        return bytes(name).length;//通过bytes强转的转换
    }
    
    function getPartName() public view returns(bytes1){
        return bytes(name)[0];
    }
    
    function changeName() public {
       // return name[0];不能够直接通过下标的方式获取string里面的内容
          
         bytes(name)[0] = 'T';
    }
}

                            通过实验，我们发现要想获取字符串中的元素，得通过bytes()进行强制转换。

                            # 2.5 地址类型

                            地址类型（address） 以太坊中的地址的长度为20字节，一字节等于8位，一共160位，所以address其实亦可以用uint160来声明。

                            地址具有 balance 属性，该属性返回账户可用的以太币数量，并具有一些用于账户间交易以太币和调用合约函数的功能。

                            它提供以下两个函数来交易以太币：

                            • transfer
                            • send

                            当向一个账户发送以太币时，更应该选择 transfer 函数而不是send函数。send 函数返回一个布尔值，具体取决于以太币发送是否成功执行，而 transfer 函数引发异常并将以太币返还给调用者。

                            它还提供了以下三个用于调用合约函数的函数：

                            • Call
                            • DelegateCall
                            • Callcode
                            address nameReg = 0x72ba7d8e73fe8eb666ea66babc8116a41bfb10e2;

nameReg.call("register", "MyName");

nameReg.call(bytes4(keccak256("fun(uint256)")), a);

                            # 2.6 枚举

                            枚举类型

                            枚举是包含一个预定义的常量值列表的值类型。它们通过值传递，每个副本都维护自己的值。不能在函数内声明枚举，并在合约的全局域命名空间内声明。预定义的常量是连续赋值的，从零开始增加整数值。如:

                            enum status {created, approved, provisioned, rejected, deleted}
